\begin{grammarEx}
	<exp> ::= <sum>
	
	<sum> ::= <prod> <sum'>
	
	<sum'> ::= \lit{\opplus{}} <prod> <sum'>
	\alt \lit{\opminus{}} <prod> <sum'>
	\alt \textemptyword{}

	<prod> ::= <pow> <prod'>
	
	<prod'> ::= \lit{\opmult{}} <pow> <prod'>
	\alt \lit{\opdiv{}} <pow> <prod'>
	\alt \textemptyword{}

	<pow> ::= <fact> <pow'>
	
	<pow'> ::= \lit{\oppow{}} <pow>
	\alt \textemptyword{}

	<fact> ::= <exp\textunderscore elem> <fact'>
	
	<fact'> ::= \lit{!}
	\alt \textemptyword{}

	<exp\textunderscore elem> ::= \lit{id}
	\alt \lit{num}
	\alt \lit{(} <exp> \lit{)}
\end{grammarEx}
